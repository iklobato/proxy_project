{% trans 'Code Examples' %}

Python

import requests

api_key = "YOUR_API_KEY"
symbol = "AAPL"

headers = {
    "Authorization": f"Bearer {api_key}"
}

response = requests.get(
    f"https://api.dadosfinanceiros.com.br/v1/quotes/{symbol}",
    headers=headers
)

data = response.json()
print(f"Price: ${data['price']}")

JavaScript

const apiKey = "YOUR_API_KEY";
const symbol = "AAPL";

fetch(`https://api.dadosfinanceiros.com.br/v1/quotes/${symbol}`, {
    headers: {
        "Authorization": `Bearer ${apiKey}`
    }
})
.then(response => response.json())
.then(data => {
    console.log(`Price: $${data.price}`);
})
.catch(error => console.error(error));
